# Crystal Keepsakes API - .htaccess
# Version: 1.0.1
# Date: 2025-11-08
# Fixed: Removed PHP-only restriction that broke Next.js

# Enable error logging (PHP only)
php_flag display_errors Off
php_flag log_errors On

# Deny access to sensitive files only
<FilesMatch "(\.env|\.log|db_connect\.php|schema\.sql)$">
    Order allow,deny
    Deny from all
</FilesMatch>

# Set memory limit for image processing (PHP only)
php_value memory_limit 256M
php_value upload_max_filesize 10M
php_value post_max_size 10M
